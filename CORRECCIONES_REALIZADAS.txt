✅ 1. includes/header.php

   CAMBIO 1: Verificación segura de variables de sesión
   ─────────────────────────────────────────────────────
   ANTES:
   $current_user = $is_logged_in ? $_SESSION['user'] : null;
   $current_profile = $is_logged_in ? $_SESSION['profile'] : null;

   AHORA:
   $current_user = $is_logged_in && isset($_SESSION['user']) ? $_SESSION['user'] : null;
   $current_profile = $is_logged_in && isset($_SESSION['profile']) ? $_SESSION['profile'] : null;

   CAMBIO 2: Verificación adicional antes de usar $current_user
   ─────────────────────────────────────────────────────────────
   ANTES:
   <?php if ($is_logged_in): ?>
       <span class="nav-link">
           <?php echo htmlspecialchars($current_user['nombre']); ?>

   AHORA:
   <?php if ($is_logged_in && $current_user): ?>
       <span class="nav-link">
           <?php echo htmlspecialchars($current_user['nombre'] ?? 'Usuario'); ?>

   BENEFICIO: Evita acceder a propiedades de null

✅ 2. usuario/dashboard.php

   CAMBIO: Uso de operador null coalescing en acceso a array
   ──────────────────────────────────────────────────────────
   ANTES:
   if (!$is_logged_in) {
       header('Location: ../login.php');
   }
   ...
   $user_id = $current_user['id'];

   AHORA:
   if (!$is_logged_in || !$current_user) {
       header('Location: ../login.php');
   }
   ...
   $user_id = $current_user['id'] ?? null;

   BENEFICIO: Validación más segura y evita warnings
